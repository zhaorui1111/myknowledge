# 音视频开发详解（AVFoundation）

## 一、AVFoundation 概述

AVFoundation 是 Apple 平台的多媒体处理框架，覆盖音视频采集、编辑、播放、导出等完整链路。它在 AVAudioEngine（音频处理）、AVCaptureSession（音视频采集）、AVPlayer（媒体播放）、AVAssetExportSession（媒体导出）等核心类上构建。

对于更高性能的需求，可以使用 VideoToolbox（硬编硬解）和 Metal（GPU 渲染）。对于实时通信，使用 AVAudioEngine + Network 编程或集成 WebRTC。

## 二、音频播放

### 2.1 AVAudioPlayer

适合播放本地音频文件，支持多种格式（mp3、aac、wav、caf）：

```swift
import AVFoundation

var audioPlayer: AVAudioPlayer?

func playSound() {
    guard let url = Bundle.main.url(forResource: "sound", withExtension: "mp3") else { return }

    do {
        audioPlayer = try AVAudioPlayer(contentsOf: url)
        audioPlayer?.delegate = self
        audioPlayer?.numberOfLoops = 0  // 0=播放一次, -1=循环
        audioPlayer?.volume = 1.0
        audioPlayer?.prepareToPlay()  // 预解码，减少延迟
        audioPlayer?.play()
    } catch {
        print("Error: \(error)")
    }
}

// AVAudioPlayerDelegate
func audioPlayerDidFinishPlaying(_ player: AVAudioPlayer, successfully flag: Bool) {
    print("Playback finished")
}
```

### 2.2 AVAudioSession

AudioSession 管理 App 的音频行为——与其他 App 音频如何共存、走哪个输出设备、是否静音：

```swift
let session = AVAudioSession.sharedInstance()

do {
    // 设置类别
    try session.setCategory(.playback,  // 播放：静音模式下也发声
                            mode: .default,
                            options: [.allowAirPlay, .allowBluetooth])
    try session.setActive(true)
} catch {
    print("Error: \(error)")
}
```

常用类别：
- `.ambient`：混音播放，受静音开关控制（默认）
- `.playback`：独占播放，不受静音影响（音乐播放器）
- `.record`：录音，阻止其他音频
- `.playAndRecord`：同时录制和播放（VoIP、语音对讲）
- `.ambient`：不中断其他音频

### 2.3 后台音频

在 Capabilities 中开启 Background Modes → Audio, AirPlay, and Picture in Picture。设置 AudioSession 类别为 `.playback`。App 进入后台后音频继续播放。

## 三、音频录制

### 3.1 AVAudioRecorder

```swift
var audioRecorder: AVAudioRecorder?

func startRecording() {
    let session = AVAudioSession.sharedInstance()
    try? session.setCategory(.record, mode: .default)
    try? session.setActive(true)

    let cacheDir = FileManager.default.urls(for: .cachesDirectory, in: .userDomainMask).first!
    let fileURL = cacheDir.appendingPathComponent("recording.m4a")

    let settings: [String: Any] = [
        AVFormatIDKey: kAudioFormatMPEG4AAC,
        AVSampleRateKey: 44100,
        AVNumberOfChannelsKey: 1,
        AVEncoderAudioQualityKey: AVAudioQuality.high.rawValue
    ]

    audioRecorder = try? AVAudioRecorder(url: fileURL, settings: settings)
    audioRecorder?.delegate = self
    audioRecorder?.record()
}

func stopRecording() {
    audioRecorder?.stop()
    audioRecorder = nil
}
```

## 四、AVAudioEngine

AVAudioEngine 提供实时音频处理管线，可以串联多个音频节点（效果器）：

```swift
let engine = AVAudioEngine()
let player = AVAudioPlayerNode()
let timePitch = AVAudioUnitTimePitch()  // 变速变调
let reverb = AVAudioUnitReverb()         // 混响

engine.attach(player)
engine.attach(timePitch)
engine.attach(reverb)

// 连接节点：player → timePitch → reverb → mixer → output
engine.connect(player, to: timePitch, format: nil)
engine.connect(timePitch, to: reverb, format: nil)
engine.connect(reverb, to: engine.mainMixerNode, format: nil)

// 加载音频文件
let audioFile = try AVAudioFile(forReading: audioURL)
player.scheduleFile(audioFile, at: nil)

// 启动引擎
try engine.start()
player.play()

// 实时调节效果
timePitch.rate = 1.5   // 1.5 倍速
timePitch.pitch = 200  // 升高 200 音分
reverb.wetDryMix = 30  // 30% 湿信号
```

### 4.1 实时音频分析

```swift
// 安装分 taps 进行频谱分析
let format = engine.mainMixerNode.outputFormat(forBus: 0)
engine.mainMixerNode.installTap(onBus: 0, bufferSize: 1024, format: format) { buffer, time in
    let level = self.computeVolume(buffer: buffer)
    DispatchQueue.main.async {
        self.updateVolumeMeter(level: level)
    }
}

func computeVolume(buffer: AVAudioPCMBuffer) -> Float {
    guard let channelData = buffer.floatChannelData?[0] else { return 0 }
    let frameLength = Int(buffer.frameLength)
    var sum: Float = 0
    for i in 0..<frameLength {
        sum += channelData[i] * channelData[i]
    }
    return sqrt(sum / Float(frameLength))
}
```

### 5.2 自定义播放器

class CustomPlayerView: UIView {
    var player: AVPlayer? {
        get { playerLayer.player }
        set { playerLayer.player = newValue }
    }

    private var playerLayer: AVPlayerLayer {
        layer as! AVPlayerLayer
    }

    override class var layerClass: AnyClass {
        return AVPlayerLayer.self

// 使用
let playerView = CustomPlayerView()
playerView.player = AVPlayer(url: url)
playerView.player?.play()

// 设置 AVPlayerLayer 的 videoGravity
if let layer = playerView.layer as? AVPlayerLayer {
    layer.videoGravity = .resizeAspectFill  // 填充

## 六、视频采集

### 6.1 AVCaptureSession

```swift
class CameraManager: NSObject {
    let session = AVCaptureSession()
    let videoOutput = AVCaptureVideoDataOutput()
    let queue = DispatchQueue(label: "camera.queue")

    func setupCamera() {
        session.beginConfiguration()
        session.sessionPreset = .high  // 1920x1080

        // 摄像头
        guard let camera = AVCaptureDevice.default(.builtInWideAngleCamera, for: .video, position: .back) else { return }
        let input = try? AVCaptureDeviceInput(device: camera)
        if session.canAddInput(input) {
            session.addInput(input)
        }

        // 视频输出
        videoOutput.setSampleBufferDelegate(self, queue: queue)
        videoOutput.videoSettings = [
            kCVPixelBufferPixelFormatTypeKey as String: kCVPixelFormatType_32BGRA
        ]
        if session.canAddOutput(videoOutput) {
            session.addOutput(videoOutput)
        }

        session.commitConfiguration()

        // 预览层
        let previewLayer = AVCaptureVideoPreviewLayer(session: session)
        previewLayer.videoGravity = .resizeAspectFill
        previewLayer.frame = view.bounds
        view.layer.insertSublayer(previewLayer, at: 0)

        // 开始采集
        session.startRunning()
    }
}

// AVCaptureVideoDataOutputSampleBufferDelegate
func captureOutput(_ output: AVCaptureOutput,
                   didOutput sampleBuffer: CMSampleBuffer,
                   from connection: AVCaptureConnection) {
    // 获取每一帧
    guard let pixelBuffer = CMSampleBufferGetImageBuffer(sampleBuffer) else { return }
    // 处理视频帧（美颜、滤镜、AI 分析等）
}
```

### 6.2 切换摄像头

```swift
func switchCamera() {
    session.beginConfiguration()

    // 移除当前输入
    for input in session.inputs {
        session.removeInput(input)
    }

    // 添加新摄像头
    let newPosition: AVCaptureDevice.Position = currentCameraPosition == .back ? .front : .back
    if let newCamera = AVCaptureDevice.default(.builtInWideAngleCamera, for: .video, position: newPosition) {
        if let newInput = try? AVCaptureDeviceInput(device: newCamera) {
            session.addInput(newInput)
            currentCameraPosition = newPosition
        }
    }

    session.commitConfiguration()
}
```

## 七、VideoToolbox 硬编硬解

VideoToolbox 提供硬件加速的编解码，性能远高于软编软解：

```swift
// H.264 硬编码
func createEncoder(width: Int, height: Int) -> VTCompressionSession? {
    var session: VTCompressionSession?
    VTCompressionSessionCreate(
        allocator: kCFAllocatorDefault,
        width: Int32(width),
        height: Int32(height),
        codecType: kCMVideoCodecType_H264,
        encoderSpecification: nil,
        imageBufferAttributes: nil,
        compressedDataAllocator: nil,
        outputCallback: { outputCallbackRefCon, sourceFrameRefCon, status, infoFlags, sampleBuffer in
            // 编码完成回调
            guard status == noErr, let sampleBuffer = sampleBuffer else { return }
            // 处理编码后的 H.264 数据
        },
        refcon: Unmanaged.passUnretained(self).toOpaque(),
        session: &session
    )

    VTSessionSetProperty(session!, key: kVTCompressionPropertyKey_RealTime, value: kCFBooleanTrue)
    VTSessionSetProperty(session!, key: kVTCompressionPropertyKey_AverageBitRate, value: 2000000 as CFNumber)
    VTSessionSetProperty(session!, key: kVTCompressionPropertyKey_MaxKeyFrameInterval, value: 30 as CFNumber)

    VTCompressionSessionPrepareToEncodeFrames(session!)

    return session
}

// 编码一帧
func encodeFrame(_ pixelBuffer: CVPixelBuffer, session: VTCompressionSession) {
    let time = CMTime(value: CMTimeValue(frameCount), timescale: 30)
    VTCompressionSessionEncodeFrame(
        session,
        imageBuffer: pixelBuffer,
        presentationTimeStamp: time,
        duration: .invalid,
        frameProperties: nil,
        sourceFrameRefcon: nil,
        infoFlagsOut: nil
    )
}
```

### 8.3 直播推流
